This procedure describes how to build pm-extras on RHEL.

# cd ..
# cp -pr pm-extras pm-extras-1.00
# tar zcf pm-extras-1.00.tar.gz pm-extras-1.00
# cp -p pm-extras-1.00.tar.gz /usr/src/redhat/SOURCES/
# cd pm-extras-1.00
# rpmbuild -ba pm-extras.spec 2>&1 | tee rpmbuild.log
